câu lệnh switch case
    ***
    cú pháp:
    switch(n){
   case value1:
            // các câu lệnh chạy nếu n == value1
            break;
        case value2:
            // các câu lệnh chạy nếu n == value2
            break;
        case value3:
            // các câu lệnh chạy nếu n == value3
            break;
        default:
            // các câu lệnh chạy khi mà n không = giá trị nào cả
    }
